TABLA USUARIOS

CREATE TABLE `olympiae`.`usuarios` (
  `id_usuario` INT NOT NULL AUTO_INCREMENT,
  `nombre_completo` VARCHAR(30) NOT NULL,
  `apellido` VARCHAR(45) NULL,
  `correo_electronico` VARCHAR(320) NOT NULL,
  `contraseña` VARCHAR(60) NOT NULL,
  `tipo_usuario` INT NOT NULL,
  PRIMARY KEY (`id_usuario`),
  UNIQUE INDEX `correo_electronico_UNIQUE` (`correo_electronico` ASC) VISIBLE)
ENGINE = InnoDB;

TABLA CURSOS

CREATE TABLE `olympiae`.`cursos` (
  `id_cursos` INT NOT NULL AUTO_INCREMENT,
  `nombre_curso` VARCHAR(45) NOT NULL,
  `descripcion` VARCHAR(1000) NOT NULL,
  PRIMARY KEY (`id_cursos`))
ENGINE = InnoDB;


TABLA DIAS_Y_HORARIOS

CREATE TABLE `olympiae`.`dias_y_horarios` (
  `id_dias_y_horarios` INT NOT NULL AUTO_INCREMENT,
  `dias_y_horarios` VARCHAR(60) NOT NULL,
  PRIMARY KEY (`id_dias-y_horarios`));


TABLA CURSOS_DIAS_HORARIOS

CREATE TABLE `olympiae`.`cursos_dias_horarios` (
  `id_cursos_dias_horarios` INT NOT NULL AUTO_INCREMENT,
  `fk_id_cursos` INT NOT NULL,
  `fk_id_dias_y_horarios` INT NOT NULL,
  PRIMARY KEY (`id_cursos_dias_horarios`),
  INDEX `fk_id_dias_y_horarios_idx` (`fk_id_dias_y_horarios` ASC) VISIBLE,
  INDEX `fk_id_cursos_idx` (`fk_id_cursos` ASC) VISIBLE,
  CONSTRAINT `fk_id_dias_y_horarios`
    FOREIGN KEY (`fk_id_dias_y_horarios`)
    REFERENCES `olympiae`.`dias-y_horarios` (`id_dias-y_horarios`)
    ON DELETE CASCADE
    ON UPDATE CASCADE,
  CONSTRAINT `fk_id_cursos`
    FOREIGN KEY (`fk_id_cursos`)
    REFERENCES `olympiae`.`cursos` (`id_cursos`)
    ON DELETE CASCADE
    ON UPDATE CASCADE);

TABLA USUARIO_CURSOS

CREATE TABLE `olympiae`.`usuario_cursos` (
  `id_usuario_cursos` INT NOT NULL AUTO_INCREMENT,
  `fk_id_usuario` INT NOT NULL,
  `fk_id_cursos_dias_horarios` INT NOT NULL,
  PRIMARY KEY (`id_usuario_cursos`),
  INDEX `fk_id_usuario_idx` (`fk_id_usuario` ASC) VISIBLE,
  INDEX `fk_id_curso_dias_horarios_idx` (`fk_id_cursos_dias_horarios` ASC) VISIBLE,
  CONSTRAINT `fk_id_usuario`
    FOREIGN KEY (`fk_id_usuario`)
    REFERENCES `olympiae`.`usuario` (`id_usuario`)
    ON DELETE CASCADE
    ON UPDATE CASCADE,
  CONSTRAINT `fk_id_curso_dias_horarios`
    FOREIGN KEY (`fk_id_cursos_dias_horarios`)
    REFERENCES `olympiae`.`cursos_dias_horarios` (`id_cursos_dias_horarios`)
    ON DELETE CASCADE
    ON UPDATE CASCADE);

SELECT
    C.nombre_curso AS NombreCurso,
    DH.dias_y_horarios AS DiaYHorario
FROM
    usuario_cursos UC
JOIN usuarios U ON UC.fk_id_usuario = U.id_usuario
JOIN cursos_dias_horarios CDH ON UC.fk_id_cursos_dias_horarios = CDH.id_cursos_dias_horarios
JOIN cursos C ON CDH.fk_id_cursos = C.id_cursos
JOIN dias_y_horarios DH ON CDH.fk_id_dias_y_horarios = DH.id_dias_y_horarios
WHERE
    U.id_usuario = 1;
